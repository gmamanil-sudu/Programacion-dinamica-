#include <iostream>
using namespace std;

int main() {
    int n, cantidad;

    //tipo de monedas
    cout << "Ingrese la cantidad de monedas: ";
    cin >> n;

    int monedas[n];

    cout << "Ingrese las monedas: ";
    for (int i = 0; i < n; i++) {
        cin >> monedas[i];
    }
    
    cout << "Ingrese la cantidad: ";
    cin >> cantidad;

    int dp[cantidad + 1];
    int usada[cantidad + 1];

    // Inicializamos la tabla con un valor grande
    for (int i = 0; i <= cantidad; i++) {
        dp[i] = 1000000;
        usada[i] = -1;
    }

    dp[0] = 0;

    // Construimos la tabla
    for (int i = 1; i <= cantidad; i++) {
        // Probamos las monedas
        for (int j = 0; j < n; j++) {
            if (monedas[j] <= i) {
                // Si encontramos una mejor solución, la guardamos
                if (dp[i - monedas[j]] + 1 < dp[i]) {
                    dp[i] = dp[i - monedas[j]] + 1;
                    // Guardamos qué moneda usamos
                    usada[i] = monedas[j];
                }
            }
        }
    }

    cout << "\nCantidad minima de monedas: " << dp[cantidad] << endl;

    // Reconstruimos una combinación válida
    cout << "Combinacion: ";
    int actual = cantidad;

    while (actual > 0) {
        cout << usada[actual];

        actual -= usada[actual];

        if (actual > 0)
            cout << " + ";
    }

    cout << "\nTabla DP: [";

    for (int i = 0; i <= cantidad; i++) {
        cout << dp[i];

        if (i < cantidad)
            cout << ", ";
    }
    cout << "]";

    return 0;
}
